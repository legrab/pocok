namespace Pocok.Modularity
{
    public interface IModuleCatalog
    {
        System.Collections.Generic.IReadOnlyList<Pocok.Modularity.ModuleDiagnostic> Diagnostics { get; }
        System.Collections.Generic.IReadOnlyList<Pocok.Modularity.ModuleDescriptor> Modules { get; }
    }
    public sealed class ModuleCatalog : Pocok.Modularity.IModuleCatalog
    {
        public ModuleCatalog(System.Collections.Generic.IReadOnlyList<Pocok.Modularity.ModuleDescriptor> modules, System.Collections.Generic.IReadOnlyList<Pocok.Modularity.ModuleDiagnostic>? diagnostics = null) { }
        public System.Collections.Generic.IReadOnlyList<Pocok.Modularity.ModuleDiagnostic> Diagnostics { get; }
        public System.Collections.Generic.IReadOnlyList<Pocok.Modularity.ModuleDescriptor> Modules { get; }
    }
    public sealed class ModuleDescriptor
    {
        public ModuleDescriptor(Pocok.Modularity.ModuleIdentity? identity, string manifestPath, string? entryAssemblyPath, bool required, Pocok.Modularity.ModuleStatus status, System.Collections.Generic.IReadOnlyList<Pocok.Modularity.ModuleDiagnostic> diagnostics) { }
        public System.Collections.Generic.IReadOnlyList<Pocok.Modularity.ModuleDiagnostic> Diagnostics { get; }
        public string? EntryAssemblyPath { get; }
        public Pocok.Modularity.ModuleIdentity? Identity { get; }
        public string ManifestPath { get; }
        public bool Required { get; }
        public Pocok.Modularity.ModuleStatus Status { get; }
    }
    public sealed class ModuleDiagnostic : System.IEquatable<Pocok.Modularity.ModuleDiagnostic>
    {
        public ModuleDiagnostic(string code, string message, Pocok.Modularity.ModuleDiagnosticSeverity severity, System.Exception? exception = null) { }
        public string Code { get; }
        public System.Exception? Exception { get; }
        public string Message { get; }
        public Pocok.Modularity.ModuleDiagnosticSeverity Severity { get; }
    }
    public enum ModuleDiagnosticSeverity
    {
        Information = 0,
        Warning = 1,
        Error = 2,
    }
    public sealed class ModuleLoadException : System.Exception
    {
        public ModuleLoadException(string message, Pocok.Modularity.ModuleCatalog catalog, System.Exception? innerException = null) { }
        public Pocok.Modularity.ModuleCatalog Catalog { get; }
    }
    public sealed class ModuleLoadOptions
    {
        public ModuleLoadOptions() { }
        public System.Collections.Generic.IList<string> Directories { get; }
        public bool IgnoreMissingDirectories { get; set; }
        public string ManifestFileName { get; set; }
        public bool SearchRecursively { get; set; }
        public System.Collections.Generic.ISet<string> SharedAssemblyNames { get; }
        public bool ThrowOnOptionalFailure { get; set; }
        public Pocok.Modularity.ModuleLoadOptions AddDirectory(string path) { }
        public Pocok.Modularity.ModuleLoadOptions ShareAssembly(string assemblyName) { }
        public Pocok.Modularity.ModuleLoadOptions ShareAssemblyContaining<T>() { }
    }
    public static class ModuleLoader
    {
        public static Pocok.Modularity.ModuleCatalog Load(Microsoft.Extensions.DependencyInjection.IServiceCollection services, Microsoft.Extensions.Configuration.IConfiguration configuration, Pocok.Modularity.ModuleLoadOptions options) { }
    }
    public sealed class ModuleManifest
    {
        public ModuleManifest() { }
        public string? ConfigurationSection { get; set; }
        public string EntryAssembly { get; set; }
        public string Id { get; set; }
        public int ManifestVersion { get; set; }
        public System.Collections.Generic.IDictionary<string, string> Metadata { get; set; }
        public bool Required { get; set; }
        public System.Collections.Generic.IList<string> SharedAssemblies { get; set; }
        public System.Collections.Generic.IList<string> SupportedArchitectures { get; set; }
        public System.Collections.Generic.IList<string> SupportedOperatingSystems { get; set; }
        public string Version { get; set; }
    }
    public static class ModuleServiceCollectionExtensions
    {
        public static Microsoft.Extensions.DependencyInjection.IServiceCollection AddPocokModules(this Microsoft.Extensions.DependencyInjection.IServiceCollection services, Microsoft.Extensions.Configuration.IConfiguration configuration, System.Action<Pocok.Modularity.ModuleLoadOptions> configure) { }
    }
    public enum ModuleStatus
    {
        Skipped = 0,
        Registered = 1,
        Failed = 2,
    }
}
