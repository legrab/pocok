namespace Pocok.Modularity
{
    public interface IServiceModule
    {
        void ConfigureServices(Microsoft.Extensions.DependencyInjection.IServiceCollection services, Pocok.Modularity.ModuleContext context);
    }
    public sealed class ModuleContext
    {
        public ModuleContext(Pocok.Modularity.ModuleIdentity identity, string baseDirectory, Microsoft.Extensions.Configuration.IConfiguration configuration, System.Collections.Generic.IReadOnlyDictionary<string, string>? metadata = null) { }
        public string BaseDirectory { get; }
        public Microsoft.Extensions.Configuration.IConfiguration Configuration { get; }
        public Pocok.Modularity.ModuleIdentity Identity { get; }
        public System.Collections.Generic.IReadOnlyDictionary<string, string> Metadata { get; }
    }
    public sealed class ModuleIdentity : System.IEquatable<Pocok.Modularity.ModuleIdentity>
    {
        public ModuleIdentity(string id, System.Version version) { }
        public string Id { get; }
        public System.Version Version { get; }
    }
}
